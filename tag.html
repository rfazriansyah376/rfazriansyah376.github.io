`; return; } // Render setiap tag container.innerHTML = sortedTags.map(tag => { // Escape single quotes untuk JSON const escapedName = tag.name.replace(/'/g, "\\'"); const postsJson = JSON.stringify(tag.posts).replace(/'/g, "\\'"); return `
${formatTagName(tag.name)}
${tag.posts.length}
${tag.posts.length > 0 ? ` Postingan terbaru:
    ${tag.posts.slice(0, 3).map(post => `
  • ${post.title}
    ${post.date}
    • `).join('')}
` : `

Belum ada postingan

`}
${tag.posts.length > 0 ? `

(totalPosts / totalTags).toFixed(1) : 0; let maxPosts = 0; allTags.forEach(tag => { if (tag.posts.length > maxPosts) { maxPosts = tag.posts.length; } }); document.getElementById('totalTags').textContent = totalTags; document.getElementById('totalPosts').textContent = totalPosts; document.getElementById('avgPosts').textContent = avgPosts; document.getElementById('maxPosts').textContent = maxPosts; } // Fungsi untuk search tags function searchTags() { const searchTerm = document.getElementById('searchInput').value.toLowerCase().trim(); if (!searchTerm) { renderTags(allTags); return; } const filteredTags = allTags.filter(tag => tag.name.toLowerCase().includes(searchTerm) || tag.posts.some(post => post.title.toLowerCase().includes(searchTerm) || (post.excerpt && post.excerpt.toLowerCase().includes(searchTerm)) ) ); renderTags(filteredTags); } // =========================================== // 5. INITIALIZATION // =========================================== document.addEventListener('DOMContentLoaded', function() { // Update statistics updateStatistics(); // Render tags pertama kali renderTags(); // Setup event listeners document.getElementById('searchInput').addEventListener('input', function() { clearTimeout(this.searchTimeout); this.searchTimeout = setTimeout(searchTags, 300); }); document.getElementById('sortSelect').addEventListener('change', renderTags); // Handle browser back button window.addEventListener('popstate', function(event) { if (event.state && event.state.tag) { document.documentElement.innerHTML = createTagPage(event.state.tag, event.state.posts); } else { location.reload(); } }); // Check jika ada halaman tag di localStorage const lastTagPage = localStorage.getItem('lastTagPage'); if (lastTagPage && window.location.pathname.includes('/tag/')) { const tag = allTags.find(t => t.name === lastTagPage); if (tag) { document.documentElement.innerHTML = createTagPage(tag.name, tag.posts); } } // Handle direct access to tag pages if (window.location.pathname.startsWith('/tag/') && window.location.pathname !== '/tag/') { const pathSegments = window.location.pathname.split('/'); const tagSlug = pathSegments[2]; const tag = allTags.find(t => createSlug(t.name) === tagSlug); if (tag) { document.documentElement.innerHTML = createTagPage(tag.name, tag.posts); } else { window.location.href = '/tag/'; } } }); // =========================================== // 6.
